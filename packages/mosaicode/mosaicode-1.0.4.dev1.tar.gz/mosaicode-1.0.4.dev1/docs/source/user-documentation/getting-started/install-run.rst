#################################
Installing and running the Mosaicode
#################################

**********
Installing
**********

.. code-block:: bash
   :linenos:

   # Go to into the Mosaicode folder that has been downloaded.
   cd mosaicode

   # Change branch
   git checkout newgui

   sudo python setup.py install

*******
Running
*******

.. code-block:: bash
   :linenos:

   mosaicode
